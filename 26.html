26. ENTER THE STRING FROM USER AND SPLITS THE FIRST WORD AND SECOND WORD AND RETURN THE START INDEX OF THE
                SECOND WORD.

            #include <stdio.h>
                #include <string.h>
                
                int splitWordsAndGetIndex(const char* str, char* firstWord, char* secondWord) {
                    int i = 0;
                    int firstWordLength = 0;
                
                    // Find the start of the first word and copy it to firstWord
                    while (str[i] != ' ' && str[i] != '\0') {
                        firstWord[i] = str[i];
                        firstWordLength++;
                        i++;
                    }
                    firstWord[i] = '\0'; // Null-terminate the first word
                
                    // Skip spaces between the words
                    while (str[i] == ' ') {
                        i++;
                    }
                
                    // Copy the second word to secondWord
                    int secondWordIndex = i;
                    while (str[i] != ' ' && str[i] != '\0') {
                        secondWord[i - secondWordIndex] = str[i];
                        i++;
                    }
                    secondWord[i - secondWordIndex] = '\0'; // Null-terminate the second word
                
                    return secondWordIndex;
                }
                
                int main() {
                    char str[100]; // Assuming a string of up to 100 characters
                    char firstWord[50];
                    char secondWord[50];
                    int secondWordIndex;
                
                    printf("Enter a string with two words: ");
                    fgets(str, sizeof(str), stdin); // Read a line of text, including spaces
                
                    secondWordIndex = splitWordsAndGetIndex(str, firstWord, secondWord);
                
                    printf("First Word: \"%s\"\n", firstWord);
                    printf("Second Word: \"%s\"\n", secondWord);
                    printf("Starting Index of Second Word: %d\n", secondWordIndex);
                
                    return 0;
                }
