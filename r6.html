//6. FIND THE SMALLEST ELEMENT IN THE INTEGER ARRAY USING RECURSION.

                #include<stdio.h>;  
  
                    #define N 5  
                      
                    int smallest(int num[], int n, int small)  
                    {  
                        if(n < 1)  
                            return small;  
                        else  
                        {  
                            if(num[n] < small)  
                                small = num[n];  
                      
                            return smallest(num, --n, small);  
                        }  
                    }  
                      
                    int main()  
                    {  
                        int a[N], i;  
                      
                        printf("Enter %d integer numbers\n", N);  
                        for(i = 0; i < N; i++)  
                            scanf("%d", &a[i]);  
                      
                        printf("Smallest Element In The Array: %d\n", smallest(a, N - 1, a[0]));  
                      
                        return 0;  
                    }
