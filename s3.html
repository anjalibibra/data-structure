3. ENTER THE STRING FROM USER AND CONVERT INTO SENTENCE CASE ?

    #include <stdio.h>
        #include<string.h>
        
        void main()
        {
            char ar[100], i;
            //int i;
            printf("enter a sentance \t");
            gets(ar);
            printf("you entered : %s" ,ar);
            for(i=0;i!='0';i++)
            {
              ar[i]=tolower(ar[i]);
            }
            ar[0]=toupper(ar[0]);
            printf(" your sentance in sentance case \t");
            printf("%s",ar);
}
