4. Enter an array from user and copy it into another array in reverse order 

            #include <stdio.h>
                void main()
                {
                    int ar[10];
                    int arr[10];
                    int i,j;
                    printf("enter an array of 10 elements");
                    for(i=0;i<10;i++)
                    {
                        scanf("%d",&ar[i]);
                    }
                    for(i=0;i<10;i++)
                    {
                        for(j=9-i;j>=0;j--)
                        {
                            arr[j]=ar[i];
                        }
                    }
                    printf(" array in reverse");
                    for(i=0;i<10;i++)
                    {
                        printf("%d \t",arr[i]);
